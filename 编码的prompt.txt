在这里，我们要处理回合数的变更逻辑（非常关键），并将 LLM 的所有原始结果保存入库，最后运行我们的 JSON 规则引擎。

Python
# main.py 的核心处理链路

# 1. 挡板：人工接管则直接中断 AI 流程
if session.is_human_takeover:
    log.info("已被人工接管，消息仅做流转录入。")
    return

# 2. 调用 LLM 并质检
ai_decision = await generate_ai_reply_with_retry(...)
session.last_llm_decision_raw = ai_decision # 🆕 完整结果直接塞进库里！

judged_next_stage = ai_decision.get("next_stage", session.current_stage)
detected_emotion = ai_decision.get("customer_emotion", "平静")
final_reply = ai_decision["reply_content"]

# ==========================================
# 3. 🎯 核心逻辑：精准计算回合数与状态突变
# ==========================================
session.total_turn_count += 1 
session.current_emotion = detected_emotion

# 如果阶段发生了跃迁：清空该阶段的回合数！
if judged_next_stage != session.current_stage:
    log.warning(f"🔄 阶段跃迁: {session.current_stage} ➡️ {judged_next_stage}")
    session.current_stage = judged_next_stage
    session.stage_turn_count = 1  # 刚进入新阶段，算是第一回合
else:
    # 如果还在原地打转，当前阶段回合数累加
    session.stage_turn_count += 1 
    log.debug(f"停留在 {session.current_stage} 已达 {session.stage_turn_count} 回合。")

# ==========================================
# 4. ⚙️ 高阶规则引擎校验 (Rule Engine)
# ==========================================
event_rules = act.event_rules if act.event_rules else []
for rule in event_rules:
    conds = rule.get("conditions", {})
    match = True
    
    # 判断情绪
    if "emotion_in" in conds and detected_emotion not in conds["emotion_in"]:
        match = False
    # 判断当前阶段
    if "stage_equals" in conds and session.current_stage != conds["stage_equals"]:
        match = False
    # 判断总回合数
    if "total_turn_count_gte" in conds and session.total_turn_count < conds["total_turn_count_gte"]:
        match = False
    # 判断阶段内回合数 (卡壳检测)
    if "stage_turn_count_gte" in conds and session.stage_turn_count < conds["stage_turn_count_gte"]:
        match = False

    # 🚨 全部条件满足，执行对应策略！
    if match:
        rule_name = rule.get('rule_name', '未命名规则')
        log.error(f"⚡ 命中高维自动化规则: 【{rule_name}】")
        
        for action in rule.get("actions", []):
            if action["type"] == "transfer_to_human":
                session.is_human_takeover = True
                # 在这里触发发送给坐席的 WebSocket 或钉钉消息
                
            elif action["type"] == "send_system_reply":
                final_reply = action["content"] # 阻断并覆盖 AI 的原话
                
            elif action["type"] == "send_material":
                final_reply += f"\n\n[🎁专属福利] {action['text']} {action['url']}"
        
        break # 命中一条高优拦截规则后，跳出循环

# 5. 落库下发
